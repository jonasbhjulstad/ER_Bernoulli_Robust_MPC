function(add_generation_executable exec_name)
    add_executable(${exec_name} ${exec_name}.cpp)
    target_link_libraries(${exec_name} PUBLIC DataFrame Common FROLS_Math FROLS_Eigen graphlite FROLS_Network ${ONEAPI_LIBS} ${HIPSYCL_LIBS})
endfunction()

endif()
